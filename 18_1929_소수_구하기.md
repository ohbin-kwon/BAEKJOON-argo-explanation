<code>

import math

a,b = map(int, input().split())

for num in range(a,b+1):
    if num == 1:
        continue
    sqrt_for_range = int(math.sqrt(num)) + 1
    for check in range(2, sqrt_for_range):
        if num % check == 0:
            break
    else:
        print(num)
