n = int(input())

for _ in range(n):
    ps_list = list(input())
    ps_sum = 0

    for i in ps_list:
        if i == '(':
            ps_sum += 1
        elif i == ')':
            ps_sum -= 1
        if ps_sum < 0:
            print('NO')
            break

    if ps_sum > 0:
        print('NO')
    elif ps_sum == 0:
        print('YES')

</code>
