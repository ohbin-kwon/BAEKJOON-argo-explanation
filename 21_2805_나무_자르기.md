## 내 풀이(시간초과)

<pre>
<code>

import sys

n , m = map(int, sys.stdin.readline().split())
trees = list(map(int, sys.stdin.readline().split()))

max_h = max(trees)

for h in range(max_h, -1, -1):
    compare_m = 0
    for tree in trees:
        if h > tree:
            continue
        else:
            compare_m += tree - h
    if compare_m == m:
        print(h)
        break
